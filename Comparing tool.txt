<script>

let mainData = [];
let compareData = [];

function readFile(file) {
    return new Promise((resolve) => {
        const reader = new FileReader();
        reader.onload = function(e) {
            let data = e.target.result
                .split(/[\s,]+/)
                .map(x => x.trim())
                .filter(x => x !== "");
            resolve(data);
        };
        reader.readAsText(file);
    });
}

async function loadFiles() {
    const mainFile = document.getElementById("mainFile").files[0];
    const compareFile = document.getElementById("compareFile").files[0];

    if (!mainFile || !compareFile) {
        alert("Please upload both files.");
        return false;
    }

    document.getElementById("progress").style.display = "block";

    mainData = await readFile(mainFile);
    compareData = await readFile(compareFile);

    return true;
}

function downloadFile(filename, content) {
    const blob = new Blob([content], { type: "text/plain" });
    const link = document.createElement("a");
    link.href = URL.createObjectURL(blob);
    link.download = filename;
    link.click();
}

async function findDuplicates() {
    if (!(await loadFiles())) return;

    const compareSet = new Set(compareData);
    const duplicates = [...new Set(mainData.filter(x => compareSet.has(x)))];

    setTimeout(() => {
        document.getElementById("progress").style.display = "none";
        downloadFile("duplicates.txt", duplicates.join("\n"));
    }, 1000);
}

async function findUnique() {
    if (!(await loadFiles())) return;

    const compareSet = new Set(compareData);
    const unique = [...new Set(mainData.filter(x => !compareSet.has(x)))];

    setTimeout(() => {
        document.getElementById("progress").style.display = "none";
        downloadFile("uniques.txt", unique.join("\n"));
    }, 1000);
}
