async function processFile() {
    const fileInput = document.getElementById('csvFile');
    const statusDiv = document.getElementById('status');
    const progress = document.getElementById('progress');

    if (!fileInput.files.length) {
        alert("Please select a CSV file.");
        return;
    }

    progress.style.display = "block";
    statusDiv.innerHTML = "Processing... Please wait.";

    const file = fileInput.files[0];
    const text = await file.text();
    const rows = text.split('\n').map(row => row.trim()).filter(row => row);

    let results = [["ZIP", "State"]];

    for (let i = 1; i < rows.length; i++) {
        const zip = rows[i].split(',')[0].trim();

        try {
            const response = await fetch(`https://api.zippopotam.us/us/${zip}`);
            
            if (response.ok) {
                const data = await response.json();
                const state = data.places[0]["state"];
                results.push([zip, state]);
            } else {
                results.push([zip, "Not Found"]);
            }

        } catch (error) {
            results.push([zip, "Error"]);
        }

        statusDiv.innerHTML = `Processing ${i}/${rows.length - 1}...`;
    }

    downloadCSV(results);
    progress.style.display = "none";
    statusDiv.innerHTML = "Done! File downloaded.";
}

function downloadCSV(data) {
    const csvContent = data.map(e => e.join(",")).join("\n");
    const blob = new Blob([csvContent], { type: "text/csv;charset=utf-8;" });
    const url = URL.createObjectURL(blob);

    const link = document.createElement("a");
    link.setAttribute("href", url);
    link.setAttribute("download", "zip_with_states.csv");
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
}
